'block' : 'none'; document.getElementById('scout-mode-single').style.display = mode==='single' ? 'block' : 'none'; document.getElementById('scout-tab-scan').style.background = mode==='scan' ? 'rgba(155,114,203,0.2)' : 'none'; document.getElementById('scout-tab-scan').style.borderColor = mode==='scan' ? '#9b72cb' : 'var(--border)'; document.getElementById('scout-tab-scan').style.color = mode==='scan' ? '#9b72cb' : 'var(--text-mid)'; document.getElementById('scout-tab-single').style.background = mode==='single' ? 'rgba(155,114,203,0.2)' : 'none'; document.getElementById('scout-tab-single').style.borderColor = mode==='single' ? '#9b72cb' : 'var(--border)'; document.getElementById('scout-tab-single').style.color = mode==='single' ? '#9b72cb' : 'var(--text-mid)'; } /* ── Auto Scout — scans top coins ── */ let autoScoutRunning = false; async function runAutoScout() { if(autoScoutRunning) return; autoScoutRunning = true; const btn = document.getElementById('auto-scout-btn'); const status = document.getElementById('auto-scout-status'); const tbody = document.getElementById('auto-scout-tbody'); const badge = document.getElementById('scout-badge'); const minMcap = parseFloat(document.getElementById('scout-mcap-filter')?.value || 200) * 1e6; const minScore = parseFloat(document.getElementById('scout-min-score')?.value || 60); if(btn) btn.textContent = '⏳ Scanning...'; if(status) status.textContent = 'Fetching market data...'; if(tbody) tbody.innerHTML = 'Scanning top coins...'; if(badge) badge.style.display = 'none'; try { // Fetch top coins list const listRes = await fetch(`${MARKET}/listings?limit=80`); const listings = await listRes.json(); const coins = (listings.coins || []).filter(c => (c.mcap||0) > minMcap && (c.volume_24h||0) > 1e6); if(status) status.textContent = `Scoring ${coins.length} coins...`; // Score each coin using momentum + available data const scored = coins.map(c => { const chg1h = c.change_1h || 0; const chg24h = c.change_24h || 0; const chg7d = c.change_7d || 0; const price = c.price || 0; const vol = c.volume_24h || 0; const mcap = c.mcap || 0; let score = 50; if(chg1h > 0) score += Math.min(chg1h * 3, 12); if(chg24h > 0) score += Math.min(chg24h * 1.5, 15); if(chg7d > 0) score += Math.min(chg7d * 0.5, 8); if(chg1h < 0) score += Math.max(chg1h * 3, -12); if(chg24h < 0) score += Math.max(chg24h * 1.5, -15); // Volume/MCap ratio bonus const vmRatio = vol / mcap; if(vmRatio > 0.1) score += 5; score = Math.max(5, Math.min(95, Math.round(score))); const bias = score >= 65 ? 'Bullish' : score <= 35 ? 'Bearish' : 'Neutral'; const bColor = score >= 65 ? '#44B828' : score <= 35 ? '#FF003F' : '#F0B429'; const atr = Math.abs(chg24h) * price / 100 * 1.5 || price * 0.02; const tp1 = score >= 50 ? price*(1 + Math.abs(chg24h)/100*1.2) : price*(1 - Math.abs(chg24h)/100*1.2); const sl = score >= 50 ? price*(1 - Math.abs(chg24h)/100*0.8) : price*(1 + Math.abs(chg24h)/100*0.8); return { ...c, score, bias, bColor, tp1, sl, chg1h, chg24h }; }) .filter(c => c.score >= minScore) .sort((a,b) => b.score - a.score) .slice(0, 15); // Check for high probability alerts (75%+) const alerts = scored.filter(c => c.score >= 75); if(alerts.length > 0 && badge) { badge.style.display = 'inline-block'; badge.textContent = `🔔 ${alerts.length} ALERT${alerts.length>1?'S':''}`; } if(!scored.length) { tbody.innerHTML = 'No coins match your filters. Try lowering the score threshold.'; if(status) status.textContent = 'No results found.'; autoScoutRunning = false; if(btn) btn.textContent = '▶ Scan All'; return; } const fmtP = (p) => p > 1000 ? '$'+p.toLocaleString('en',{maximumFractionDigits:0}) : p > 1 ? '$'+p.toFixed(2) : '$'+p.toFixed(4); tbody.innerHTML = scored.map((c,i) => ` ${c.score>=75?'🔔 ':''}${(c.symbol||c.name||'').replace('USDT','').replace('USD','')} ${c.score}% ${c.bias} ${fmtP(c.tp1)} ${fmtP(c.sl)} `).join(''); if(status) status.textContent = `Found ${scored.length} setups · ${alerts.length} high-probability alerts · Click row for details`; } catch(err) { console.error('Auto Scout error:', err); if(status) status.textContent = 'Error fetching data. Check connection.'; if(tbody) tbody.innerHTML = 'Connection error. Retry.'; } autoScoutRunning = false; if(btn) btn.textContent = '▶ Scan All'; } function showScoutDetail(data) { const detail = document.getElementById('auto-scout-detail'); if(!detail) return; const fmtP = (p) => p > 1000 ? '$'+p.toLocaleString('en',{maximumFractionDigits:0}) : p > 1 ? '$'+p.toFixed(2) : '$'+p.toFixed(4); detail.style.display = 'block'; detail.innerHTML = `
${data.sym} · ${data.bias} · AI Score ${data.score}%
Entry: ${fmtP(data.price)}
1H: ${data.chg1h>=0?'+':''}${parseFloat(data.chg1h).toFixed(2)}%
TP1: ${fmtP(data.tp1)}
24H: ${data.chg24h>=0?'+':''}${parseFloat(data.chg24h).toFixed(2)}%
SL: ${fmtP(data.sl)}
Side: ${data.score>=50?'Long':'Short'}
${data.bias} momentum · Score ${data.score}% · ${data.score>=75?'⚡ High probability setup!':data.score>=65?'Watch for entry':'Neutral — wait for confirmation'}
`; } /* ── Auto-scan every 5 minutes ── */ setInterval(() => { if(document.getElementById('panel-scout')?.classList.contains('open')) { runAutoScout(); } }, 5 * 60 * 1000);
